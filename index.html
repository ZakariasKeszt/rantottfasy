int biggest_3digit_num(int *p){

int ered = 0; //eredmeny valtozo

while(*p){ //ameg a tomb nem nulla

if(*p > 99 && *p < 1000 && *p > ered){ //haromjegyu feltetel es nagyobb mint az elozo ertek

ered = *p;

}

p++; //pointer novelese

}

return ered;

}

int char_count(char *p){

int count = 0; //szamlalo

while(*p){ //ameg a tomb nem nulla

if(*p == 'o' || *p == 'k'){ //tomb eleme o vagy k

count++;

}

p++; //pointer novelese

}

return count;

}
